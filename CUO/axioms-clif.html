(cl:comment 'In this document x,y,z are used for instances, c,d,e for
universals, r,r1,r2 for relations, and t,t1,t2 for temporal instants. For the most part the instance/universal distinction of relation arguments is defined or definable for each relation.')

(cl:comment 'Every type has an instantiation, every instance has a type'
(forall ((c type))
      (or (exists (x) (or (instance_of x c)))
          (exists (x (t Time)) (instance_of x c t))))    
(forall ((x instance))
      (or (exists (c) (or (instance_of x c)))
          (exists (c (t Time)) (instance_of x c t)))))

(cl:comment 'Constraints on type_type and instance_instance relations'
(forall (r relation)(iff (type_type r)
                    (forall (c d (t Time))
                          (and
                           (if (r c d)
                               (and (type c)
                                  (type d)))
                           (if (r c d t)
                               (and (type c)
                                  (type d)))))))
(forall (r relation)(iff (instance_instance r)
                    (forall (x y (t Time))
                          (and
                           (if (r x y)
                               (and (instance x)
                                  (instance y)))
                           (if (r x y t)
                               (and (instance x)
                                  (instance y)))))))
(forall (r relation)(iff (instance_type r)
                    (forall (x y (t Time))
                          (and
                           (if (r x y)
                               (and (instance x)
                                  (type y)))
                           (if (r x y t)
                               (and (instance x)
                                  (type y)))))))
(forall (r) (if (or (instance_instance r) (type_type r)
                (instance_type r)) (relation r))))

(cl:comment 'temporal relations not considered here for
          symmetry,transitivity, and reflexivity')
(cl:comment 'relation symmetry, asymmetry, nonsymmetry, and antisymmetry'
(forall (x y (r relation))       
      (iff (symmetric r)
           (if (r x y) (r y x))))
(forall (x y (r relation))       
      (iff (asymmetric r)
           (if (r x y) (not (r y x)))))
(forall (x y (r relation))
      (iff (nonsymmetric r)
           (and (not (symmetric r)) (not (asymmetric r)))))
(forall (x y (r relation))       
      (iff (antisymmetric r)
           (if (r x y) (and (r y x) (= x y))))))

(cl:comment 'relation transitivity, intransitivity, and nontransitivity'
(forall (x y z (r relation))
      (iff (transitive r)
           (if (and (r x y) (r y z))
             (r x z))))
(forall (x y z (r relation))
      (iff (intransitive r)
           (if (and (r x y) (r y z))
             (not (r x z)))))
(forall (x y (r relation))
      (iff (nontransitive r)
           (and  (not (transitive r)) (not (asymmetric r))))))

(cl:comment 'relation reflexivity, irreflexivity, and nonreflexivity'
(forall (x y (r relation))
      (iff (reflexive r)
           (if (r x y) (r x x))))
(forall (x y (r relation))
      (iff (irreflexive r)
           (if (r x y) (not (r x x)))))
(forall (x y (r relation))
      (iff (nonreflexive r)
           (and (not (reflexive r)) (not (asymmetric r))))))

(cl:comment 'is_a relation is between individuals (particulars) and types (universals)'
(and (type_type is_a) (transitive is_a) (antisymmetric is_a) (reflexive is_a)))

(and (instance_type instance_of) (intransitive instance_of)
     (antisymmetric instance_of) (reflexive instance_of))
(and (type_type part_of) (instance_instance part_of)
     (transitive part_of) (antisymmetric part_of) (reflexive part_of))

(cl:comment 'subrelation'
(forall ((r1 relation)(r2 relation) x y)
      (if (and (subrelation r1 r2)
             (r1 x y))
          (r2 x y)))
(forall ((r1 relation)(r2 relation)(t Time) x y)
      (if (and (subrelation r1 r2)
             (r1 x y t))
          (r2 x y t))))

(cl:comment 'Proper subrelation'
(forall ((pr relation)(r relation))
      (if (proper_subrelation pr r)
          (and (irreflexive pr)
             (subrelation pr r)))))

(cl:comment 'all_some_all_times'
(forall ((tr type_type)(ir instance_instance)(t Time) x y c d)
      (if (all_some_all_times tr ir)
          (and (tr c d)
             (if (instance_of x c t)
                 (exists (y) (and (instance_of y d t))
                       (ir x y t)))))))

(cl:comment 'all_some_all_times example'
(all_some_all_times part_of_some part_of))
                            
(cl:comment 'all_some'
(forall ((tr type_type) (ir instance_instance) x y c d)
      (if (all_some tr ir)
          (and (tr c d)
             (if (instance_of x c)
                 (exists (y) (and (instance_of y d))
               (ir x y)))))))

(cl:comment 'all_only'
(forall ((tr type_type)(ir instance_instance) x y c d)
      (iff (all_only tr ir)
          (and (tr c d)
             (if (instance_of x c)
                 (not (exists (y) (and (not (instance_of y d))
                                 (ir x y)))))))))

(cl:comment 'Propogation of all_only relations under(over) is_a'
(forall ((tr type_type)(ir instance_instance) c d e)
      (and (tr c d)
           (if (is_a d e)
             (is_a c e))))
(forall ((tr type_type)(ir instance_instance) c d e)
      (and (is_a c d)
           (if (tr d e)
             (tr c e)))))


(cl:comment 'AI,AU,AM,TM,TU extracted from
          "Individuals, Univerisals, Collections: On the Foundational Relations of Ontology", Bittner et al.')

(cl:comment 'AI2 - if two universals share an instance x at some time t then the universals taxonomically overlap.'
(forall (x c d t) (if (and (instance_of x c t)
                     (instance_of x d t))
                  (overlaps c d))))

(cl:comment 'AI3 - if x is a instance of a universal at t then x exists at t'
(forall (x c t) (if (instance_of x c t)
                (exists_at x t))))

(cl:comment 'AI4 - every universal is instantiated at some time'
(forall (c Universal) (exists (x (t Time)) (instance_of x c t))))

(cl:comment 'AI5 - every time at which an individual exists it is an instance of some universal'
(forall (x (t Time)) (if (exists_at x t)
                   (exists (c) (instance_of x c t)))))

(cl:comment 'AI6 - If two univerals have the same instances at every time then they are identical'
(forall (x (t Time)) (if
                   (iff (instance_of x c t)
                        (instance_of x d t))
                   (= c d))))

(cl:comment 'Relation definitions for taxonomic inclusion, IUC paper, Sec. 3.2')

(and (type_type proper_is_a) (type_type overlaps))

(cl:comment 'proper subuniversal'
(forall (c d)(iff (proper_is_a c d)
               (and (is_a c d)
                  (not (= c d))))))

(cl:comment 'taxonomic overlap'
(forall (c d)(iff (overlaps c d)
              (exists ((e Universal))
                    (and (is_a e c)
                         (is_a e d))))))

(cl:comment 'Definition for universal root'
(forall ((c Universal))
      (iff (Root c)
           (exists ((g Universal)) (is_a g c)))))

(forall (c d)

(forall (c d e)

(cl:comment 'Evolves from among Universals'
(forall (c d x (t Time))
      (iff (evolves_from c d)

(cl:comment 'instance_of relation. Continuants have temporal parts, occurents do not.'
(forall (x c (t Time))
      (if (instance_of x c t)
          (is_a c Continuant)))
(forall (x c)
      (if (instance_of x c)
          (is_a c Occurent))))

(cl:comment 'is_a relation'
(forall (x (c Continuant) (d Continuant))
      (if (is_a c d)
          (exists ((t Time))
                (if (instance_of x c t)
                  (instance_of x d t)))))
(forall (x (c Process) (d Process))
      (if (is_a c d)
          (if (instance_of x c)
            (instance_of x d)))))

(cl:comment 'class level part_of relation and has_part (reciprocal_relation)'
(forall ((x Continuant)(c Continuant)(d Continuant) (t Time))
      (iff (part_of c d)
           (if (instance_of x c t)
             (exists (y Continuant) (and (instance_of y d t)
                                   (part_of x y t))))))
(forall ((x Process)(p Process)(q Process))
      (iff (part_of p q)
           (if (instance_of x p)
             (exists (y Process) (and (instance_of y q)
                                     (part_of x y))))))
(forall ((x Continuant)(c Continuant)(d Continuant) (t Time))
      (iff (has_part c d)
           (if (instance_of x c t)
             (exists (y Continuant) (and (instance_of y d t)
                                   (has_part x y t))))))
(forall ((x Process)(p Process)(q Process))
      (iff (has_part p q)
           (if (instance_of x p)
             (exists (y Process) (and (instance_of y q)
                                     (has_part x y)))))))

(cl:comment 'instance level part_of'
(forall (x y s c d t)
      (if (part_of x y t)
          (and
           (instance_of x Continuant)
           (instance_of y Continuant)
           (instance_of s Temporal_Instant))))
(forall (x y s c d)
      (if (part_of x y t)
          (and
           (instance_of x Occurent)
           (instance_of y Occurent)))))

(cl:comment 'instance level proper_part_of and has_proper_part'
(forall ((x Continuant) (y Continuant) (t Time))
      (iff (proper_part_of x y t) 
          (and (part_of x y t)
             (not (part_of y x t)))))
(forall ((p Process) (q Process))
      (iff (proper_part_of p q) 
          (and (part_of p q)
             (not (part_of q p)))))
(forall ((x Continuant) (y Continuant) (t Time))
      (iff (has_proper_part x y t) 
          (and (has_part x y t)
             (not (has_part y x t)))))
(forall ((p Process) (q Process))
      (iff (has_proper_part p q) 
          (and (has_part p q y)
             (not (has_part q p))))))

(cl:comment 'class level integral_part_of and has_integral_part'
(forall (c d) (iff (integral_part_of c d)
               (and (part_of c d)
                  (has_part d c))))
(forall (c d) (iff (has_integral_part c d)
               (and (part_of c d)
                  (has_part d c))))
(and (transitive integral_part_of) (reflexive integral_part_of)
     (antisymmetric integral_part_of))
(and (transitive has_integral_part) (reflexive has_integral_part)
     (antisymmetric has_integral_part)))

(cl:comment 'instance level exists_at'

           (exists (c) (instance_of x c t))))
(forall (x (t Time))
      (iff (exists_at x t)
